Remarks

Suppose you have a simple model like this:

public class MyModel
{
public int Number { get; set; }
public string Name { get; set; }
}

And according to your rules, the number must be between 1 and 99, and the name must be "Brazil". You can express this in a rule like this:

public class MyRule1 : RuleFor&lt;MyModel>
{
public MyRule1()
    : base(&quot;RN-1&quot;, &quot;My rule 1&quot;)
{ }

protected async override Task&lt;RuleCheckResult> CheckAsync(MyModel target)
{
    if (target.Number &lt; 1 || target.Number > 99)
        return RuleCheckResult.Fail;

    if (target.Name != &quot;Brazil&quot;)
        return RuleCheckResult.Fail;

    return RuleCheckResult.Success;
}
}

Now that you have a defined rule, just use it in your methods and business objects like this:

public class MyBusinessClass
{
public void MyMethod1(MyModel model)
{
    new MyRule1().Ensure(model);
}
}
